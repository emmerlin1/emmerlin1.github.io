* Erstellen Sie mit dem Editor nano eine Datei mit dem Namen '**readpipe.sh**' mit folgendem Inhalt:

{%ace edit=true, lang='sh'%}

#!/bin/bash

pipe=/tmp/testpipe

trap "rm -f $pipe" EXIT

if [[ ! -p $pipe ]]; then
    mkfifo $pipe
fi

while true
do
    if read line <$pipe; then
        if [[ "$line" == 'quit' ]]; then
            break
        fi
        echo $line
    fi
done

echo "Reader exiting"

{%endace%}
